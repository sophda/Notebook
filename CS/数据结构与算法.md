class Solution {

        unordered_map<string, vector<string>> map_;
        for(auto& str : strs)
        {
            string key = str;

class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        if(nums.empty()) return 0;
        unordered_set<int> set_;
        for(auto i : nums){
            set_.insert(i);
        }
        int max_len = 1;
        int cur_num = 0;
        int cur_len = 0;
        for(int a : set_){
            if(!set_.count(a-1)) {
                cur_num = a;
                cur_len = 0;
                while(set_.count(cur_num++)){
                    cur_len++;
                }
                max_len = max(max_len,cur_len);
            }
        }
        return max_len;
        
    }
};

class Solution {

        int left = 0;
        int right = height.size() - 1;
        int max_area = 0;
        int temp = 0;
        while (left != right) {

## 15.三数之和

[15. 三数之和 - 力扣（LeetCode）](https://leetcode.cn/problems/3sum/?envType=study-plan-v2&envId=top-100-liked)

![image-20250609234645589](src/image-20250609234645589.png)

```c++
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> result;

        sort(nums.begin(),nums.end());

        for(int first = 0; first<nums.size();first++) {
            if(first > 0 && nums[first]==nums[first-1]){
                continue;
            }
            int third = nums.size()-1;

            for (int second = first+1; second<nums.size();second++) {
                if(second>first+1 && nums[second] == nums[second-1]) {
                    continue;
                }
                while(third>second && nums[first]+nums[second]+nums[third]>0) {
                    third--;
                }
                if(second==third) {
                    break;
                }

                if (nums[first]+nums[second]+nums[third]==0) {
                    result.push_back({nums[first],nums[second],nums[third]});
                }
            }
        }
        return result;
    }
};
```

这个主要考虑的是需要使用三个循环来解决三数之和为0的情况，但是三重循环时间复杂度为$n^{3}$,所以在2、3重循环的时候，把2重循环保留，三重循环为指向尾部的指针。

---

需要注意的是初始化third值的地方：这句话放到了二维循环的外面，为什么要这样做呢？

```
int third = nums.size()-1;
```

因为nums数组是经过排序之后的，在第二重循环中，second是递增的，所以third是肯定递减的，而且并不需要从尾部开始重新查找。比如在second=3、third=6的时候满足情况，下一轮循环second=4，那么third必然小于6。

**这条如果放在二维循环里面，数据量小时并不会有什么后果，如果数据量比较大会超时**

## 3.无重复字符的最长字串

[3. 无重复字符的最长子串 - 力扣（LeetCode）](https://leetcode.cn/problems/longest-substring-without-repeating-characters/description/?envType=study-plan-v2&envId=top-100-liked)

这道题主要是使用滑动窗口解决，那么窗口在滑动的过程中，需要不断检查新加进来的元素在窗口中有没有存在过，因此这个时候就不能用vector了，而是使用带有查找功能的哈希表即`unordered_set`.

unordered_set不支持放重复的元素，刚好这个题目要求的就是无重复的最长字串，用来做窗口岂不美哉~ 或许你想说，窗口移动会insert新的元素，这不会跟之前的冲突了吗？因为是不重复的最长子串，所以insert之前肯定会count一下看看有没有这个字符的。。

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> window;
        int rp = 0;
        int ans = 0;
        for(int i = 0; i<s.size();i++) {
            if(i!=0) {
                window.erase(s[i-1]);
            }
            while(rp<s.size() && !window.count(s[rp])) {
                window.insert(s[rp]);
                rp++;
            }
            ans = max(ans,rp-i);
        }
        return ans;
    }
};
```

关键点在于**滑动窗口**，这就要求，要有两个指针分别管理着窗口的左右边界，有一个unordered_set管理窗口内容。

- 右指针移动一位，set.insert()
- 左指针移动一位，set.erase()

## 53.最大子数组和

[53. 最大子数组和 - 力扣（LeetCode）](https://leetcode.cn/problems/maximum-subarray/description/?envType=study-plan-v2&envId=top-100-liked)

![image-20250609001653422](src/image-20250609001653422.png)

这道题目是动态规划相关的，也可以用分治算法来解决。鉴于动态规划代码量比较少，先来用这个方法。

动态规划比较重要的点在于，转移方程，类似于数学归纳法。

**最大子数组和肯定是上一个最大子数组和加上新元素之后，看看哪个大**，因此，转移方程为：

![image-20250609003056400](src/image-20250609003056400.png)

i表示元素的下标，为什么是f(i-1)+nums[i]和nums[i]进行比较的呢？为什么不是f(i-1)+nums[i]和f(i-1)进行比较的呢?

考虑这种情况：1，当f(i-1)是正的，这样子比没问题，因为f(i-1)+nums[i]肯定比f(i-1)要大；2.如果f(i-1)是负数，这时候就会“拖后腿”，还不如舍弃，取新的值作为最大值

```c++
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int ans_max = -65536;
        int pre = 0;
        int n = nums.size();
        for(int i = 0; i < n; i++) {
            pre = max(pre+nums[i], nums[i]);
            ans_max = max(pre, ans_max);
        }
        return ans_max;
    }
};
```

ans_max要取个非常小的值。



## 56.合并区间

[56. 合并区间 - 力扣（LeetCode）](https://leetcode.cn/problems/merge-intervals/description/?envType=study-plan-v2&envId=top-100-liked)

---

这道题目，记住一下怎么用，相关api以及二维数组怎么调整的某一个值。

- vector.back()，表示vector的最后一个元素。
- sort()排序二维数组是看第一个元素。

```c++
class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        vector<vector<int>> ans;
        int n = intervals.size();
        if (n==0) return ans;
        sort(intervals.begin(),intervals.end());
        for(int i = 0;i<n;i++) {
            if(i==0 || intervals[i][0]>ans.back()[1]) {
                ans.push_back({intervals[i][0],intervals[i][1]});
            }
            else {
                ans.back()[1] = max(ans.back()[1], intervals[i][1]);
            }
        }
        return ans;
    }
};
```

        int n = nums.size();
        int tail = n-1; int head = 0;
        vector<int> head_v;
        vector<int> tail_v;
        for(int i = 0; i<n-1; i++) {

                head_v.push_back(nums[i]);
            }
            else{
                head_v.push_back(head_v.back()*nums[i]);

            }
        }
        for(int j = n-1; j>0;j--) {
            if(j==n-1) {
                tail_v.push_back(nums[j]);
            }
            else{
                tail_v.push_back(tail_v.back()*nums[j]);

                ans[k] = (tail_v[n-1-1]);

            else if(k==n-1) {
                ans[k] = (head_v[n-1-1]);
            }
            else {
                ans[k] = (head_v[k-1]*tail_v[n-k-2]);

            }
        }
        return ans;
    }
};

class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        unordered_set<int> col;
        unordered_set<int> row;
        for(int i=0;i<m;i++) {
            for(int j=0;j<n;j++){
                if(matrix[i][j]==0) {
                    row.insert(i);
                    col.insert(j);
                }
            }
        }
        for(int i=0;i<m;i++) {
            for(int j=0;j<n;j++) {
                if(row.count(i) || col.count(j)) {
                    matrix[i][j] = 0;
                }
            }
        }
        
    }
};

            return list2;

            return list1;

- 先求出左右节点的深度，那么该节点的深度就是两者最大值。

- 因为是遍历了每一个节点，题目说明最长路径可能不会经过root节点，那么每个遍历的节点都有可能是最大路径的根节点，也就是每个节点遍历的时候都要更新一下最大节点数。

 *     TreeNode *left;

 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}

 * };
 */
class Solution {
public:
    int ans;
    int depth(TreeNode* rt) {
        if (rt == nullptr) return 0;
        int l = depth(rt->left);
        int r = depth(rt->right);
        int max_dep = max(l,r)+1;
        // ans = max(ans,max_dep);
        ans = max(ans,l+r+1);
        return max_dep;
    }
    int diameterOfBinaryTree(TreeNode* root) {
        ans = 1;
        depth(root);
        return ans-1;
        
    }
};

        // if(rt->val == targetSum) return ++sum;

 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;

class Solution {

public:
    bool after(TreeNode* rt, TreeNode* p, TreeNode* q) {

## 46.全排列

[46. 全排列 - 力扣（LeetCode）](https://leetcode.cn/problems/permutations/?envType=study-plan-v2&envId=top-100-liked)

![image-20250623235815512](src/image-20250623235815512.png)

---

全排列是在回溯的基础上增加了一个访问标志：回溯的**回**，可以理解成恢复之前的状态。

```cpp
class Solution {
public:
    vector<int> used;
    vector<vector<int>> ans;
    vector<int> path;
    void backtracking(vector<int>& nums, vector<int>& used) {
        if(nums.size() == path.size()) {
            ans.push_back(path);
            return;
        }

        for(int i = 0; i < nums.size(); i++) {
            if(used[i]) continue;

            path.push_back(nums[i]);
            used[i] = true;
            backtracking(nums, used);
            path.pop_back();
            used[i] = false;
        }

    }

    vector<vector<int>> permute(vector<int>& nums) {
        vector<int> used(nums.size(), false);
        backtracking(nums, used);
        return ans;
    }
};
```

- 回溯是一种暴力求解，每一个回溯问题都可以抽象成一个**二叉树**，既然是二叉树，就可以用层来分析每一次递归在干什么。
- 比如说[1,2,3]这三个数，第一层（第一次调用递归函数）是path中放了1；
- 然后第二层（第二次调用递归）按照used将2放到pah中；
- 执行第三层（第三次调用递归）按照used将3放到path
- 第四层，第四次调用递归， 将4放到path
- 第五层，达到终止条件，ans存path，跳到第四层，取消标记，然后一路回溯到第一层的for循环第二个数，即path中放第一个数为2，依次循环。

    void backtracking(vector<int>& candidates, int len, int target, int start_id) {
        int sum = std::accumulate(path.begin(), path.end(), 0);
        if(sum > target) return;
        if(path.size() == len) {
            if(sum == target) {
                ans.push_back(path);
            }
            return;
        }

        for(int i = start_id; i < candidates.size(); i++) {

            backtracking(candidates, len, target, i);
            path.pop_back();
        }
    }

        sort(candidates.begin(), candidates.end());

            backtracking(candidates, i, target, 0);
        }

    }

                left = mid+1;
            }
            else if(target < x) {
                right = mid-1;
            }
            else{
                return true;
            }
        }
        return false;
    }
};

class Solution {
public:
    int findMin(vector<int>& nums) {
        int left=0, right=int(nums.size())-1, mid = 0;
        while(left <= right) {
            mid = left + (right-left)/2;
            if(nums[mid] < nums[right]) {
                right = mid;
            } else {
                left = mid+1;
            }
        }
        return mid;
    }

    int binarySearch(vector<int>& nums, int left, int right, int target) {
        int ans = -1;
        while(left<=right) {
            int mid = (left+right)/2;
            if(target<nums[mid]) {
                right = mid-1;
            } else if (target > nums[mid]) {
                left = mid+1;
            } else {
                ans = mid;
                return ans;
            }
        }
        return ans;
    }

    int search(vector<int>& nums, int target) {
        int id = findMin(nums);
        int ans = -1, n = int(nums.size())-1;
        if(nums[n]>=nums[0]) {
            return binarySearch(nums,0,n,target);
        }
        if(target>=nums[0]) {
            ans = binarySearch(nums, 0, id-1, target);
        } else {
            ans = binarySearch(nums, id, int(nums.size())-1, target);
        }
        return ans;
    }
};
